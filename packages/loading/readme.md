# @fluentliteui/loading

install
```
yarn add @fluentliteui/loading
```

usage
```jsx
import { loading } from "@fluentliteui/loading";
import "@fluentliteui/loading/dist/index.css";

function App() {
  <Loading loading={true}>
    <p>1</p>
    <p>2</p>
    <p>3</p>        
  </Loading>
}
```
